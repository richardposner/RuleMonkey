# Simple A+A reaction

begin parameters
t_end		10
n_steps         100

A_tot		1000

kp		0.001
km		0.5
end parameters

begin molecule types
A(a)
end molecule types

begin species
A(a)		A_tot
end species

begin reaction rules
A(a) + A(a) <-> A(a!1).A(a!1) kp,km
end reaction rules

begin observables
Species A   A(a)
Species AA   A(a!1).A(a!1)
end observables

simulate_rm({t_end=>t_end,n_steps=>n_steps});
